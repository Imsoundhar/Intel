CREATE OR REPLACE TABLE ORDER_DATABASE_E1.ORDER_BASE.T_SALES_ORDER_HEADER (
    ORDER_ID VARCHAR(10) PRIMARY KEY,
    ORDER_DATE DATE,
    STATUS VARCHAR(10),
    CUSTOMER_ID VARCHAR(10),
    ITEM_COUNT INTEGER,
    TOTAL_AMOUNT NUMBER(12,2),
    DISCOUNT NUMBER(10,2),
    CREATE_DATE DATE,
    CREATE_USER VARCHAR(10),
    COUPON_CODE VARCHAR(10)
);

CREATE OR REPLACE TABLE ORDER_DATABASE_E1.ORDER_BASE.T_SALES_ORDER_DETAIL (
    ORDER_ID VARCHAR(10),
    ORDER_DATE DATE,
    CUSTOMER_ID VARCHAR(10),
    ITEM_ID VARCHAR(10),
    QTY INTEGER,
    TOTAL_AMOUNT NUMBER(12,2),
    PRICE NUMBER(12, 2),
    CREATE_DATE DATE,
    CREATE_USER VARCHAR(10),
    PRIMARY KEY (ORDER_ID, ITEM_ID), -- Assuming composite key
    FOREIGN KEY (ORDER_ID) REFERENCES T_SALES_ORDER_HEADER(ORDER_ID)
);

CREATE OR REPLACE TABLE ORDER_DATABASE_E1.ORDER_BASE.T_CUSTOMER (
    CUSTOMER_ID VARCHAR(10),
    ADDRESS VARCHAR(50),
    PHONE_NO VARCHAR(10),
    CATEGORY VARCHAR(10),
    STATUS VARCHAR(10),
    TOTAL_ORDERS INTEGER,
    TOTAL_INVOICE_AMOUNT NUMBER(12,2),
    LOYALTY_POINTS NUMBER(12,2),
    CREATE_DATE	DATE,
    CREATE_USER	VARCHAR(10),
    PRIMARY KEY (CUSTOMER_ID)
);

CREATE OR REPLACE TABLE ORDER_DATABASE_E1.ORDER_BASE.T_PROMOTION (
    PROMOTION_ID VARCHAR(10) PRIMARY KEY,
    CUSTOMER_CATEGORY VARCHAR(10),
    PROMOTION_TYPE VARCHAR(10),
    PROMOTION_VALUE NUMBER(10,2),
    COMMENT VARCHAR(20) -- Assuming this could hold more details
);

CREATE OR REPLACE TABLE ORDER_DATABASE_E1.ORDER_BASE.T_INVOICE (
    ORDER_ID VARCHAR(10),
    ORDER_DATE	DATE,
    INVOICE_DATE	DATE,
    TAX_AMOUNT	NUMBER(10,2),
    TOTAL_AMOUNT NUMBER(12,2),
    DISCOUNT NUMBER(10,2),
    FINAL_INVOICE_AMT NUMBER(12,2),
    CUSTOMER_ID	VARCHAR(10),
    CREATE_DATE DATE,
    CREATE_USER VARCHAR(10),
    FOREIGN KEY (ORDER_ID) REFERENCES T_SALES_ORDER_HEADER(ORDER_ID)
);
